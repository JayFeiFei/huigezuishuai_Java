1. 完成以下操作
  1.	创建英雄类
    属性: 名字, 攻击力, 移动速度
    行为: 攻击, 移动
    攻击力随机范围[1-10], 移动速度随机范围[1-5]
    名字可以自己指定
  1. 创建一个ArrayList, 存10个英雄对象
  3. 删除下标为奇数的英雄
  4. 计算攻击力的平均值
  5. 获取移动速度最高的英雄, 如果有多个, 返回多个

  ```java
  Hero.java
  public class Hero {
      //名字, 攻击力, 移动速度
      private String name;
      private int atk;
      private int speed;
  
      public Hero() {
      }
  
      public Hero(String name, int atk, int speed) {
          this.name = name;
          this.atk = atk;
          this.speed = speed;
      }
  
      public String getName() {
          return name;
      }
  
      public void setName(String name) {
          this.name = name;
      }
  
      public int getAtk() {
          return atk;
      }
  
      public void setAtk(int atk) {
          this.atk = atk;
      }
  
      public int getSpeed() {
          return speed;
      }
  
      public void setSpeed(int speed) {
          this.speed = speed;
      }
  
      @Override
      public String toString() {
          return "Hero{" +
                  "name='" + name + '\'' +
                  ", atk=" + atk +
                  ", speed=" + speed +
                  '}';
      }
  
      // 攻击, 移动
      public void attack() {
          System.out.println("全军出击!");
      }
  
      public void move() {
          System.out.println("移动!");
      }
  }
  
  Main.java
  public class Main {
      public static void main(String[] args) {
          
          String[] names = {"凯隐", "洛", "霞", "卡蜜尔", "艾翁", "克烈", "塔莉垭", "索尔", "烬", "俄洛伊", "千珏", "塔姆", "艾克", "巴德", "雷克赛", "卡莉丝塔", "阿兹尔", "纳尔", "布隆", "维克兹", "亚索", "金克丝"};
  
          ArrayList<Hero> heroes = new ArrayList<>();
  
          Random random = new Random();
          for (int i = 0; i < 10; i++) {
              String name = names[random.nextInt(names.length)];
              int atk = random.nextInt(10) + 1;
              int speed = random.nextInt(5) + 1;
              Hero hero = new Hero(name, atk, speed);
              heroes.add(hero);
              System.out.println(hero);
          }
  
          for (int i = heroes.size() - 1; i >= 0; i--) {
              if (i % 2 == 1) {
                  heroes.remove(i);
              }
          }
  
          System.out.println("删除后:");
          for (Hero hero : heroes) {
              System.out.println(hero);
          }
      }
  }
  ```

   HashMap<String, Integer> hashMap = new HashMap<>();
   hashMap.put("小明", 20);
   hashMap.put("小花", 99);
   hashMap.put("小刚", 33);
   hashMap.put("小红", 76);
   hashMap.put("小智", 5);
   hashMap.put("小龙", 100);
   hashMap.put("小雅", 98);
   hashMap.put("小胖", 2);
   
   ArrayList<Integer> scores = new ArrayList<>(hashMap.values());
   System.out.println(scores);
   
   for (int i = 0; i < scores.size() - 1; i++) {
       for (int j = 0; j < scores.size() - 1 - i; j++) {
           if (scores.get(j) < scores.get(j + 1)) {
               scores.add(j, scores.remove(j + 1));
           }
       }
   }
   System.out.println(scores);
   
   for (Map.Entry<String, Integer> entry : hashMap.entrySet()) {
   
       if (entry.getValue().equals(scores.get(0)) || entry.getValue().equals(scores.get(1)) || entry.getValue().equals(scores.get(2))) {
           System.out.println(entry.getKey());
       }
   }
